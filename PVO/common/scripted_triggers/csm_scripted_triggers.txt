can_reach_irish_mounts_trigger = {
	OR = {
		AND = {
			NOR = {
				has_global_flag = csm_plus_detected
				has_global_flag = csm_hip_detected
			}
			primary_title = {
				ROOT = {
					any_realm_province = {
						OR = {
							de_jure_liege_or_above = PREVPREV
							ROOT = {
								capital_scope = {
									province = PREVPREV 
								}
							}
						}	
						NOT = { distance = { where = event_target:ireland_mount_province value = 350 } }
					}
				}
			}
		}	
		AND = {
			has_global_flag = csm_plus_detected
			primary_title = {
				ROOT = {
					any_realm_province = {
						OR = {
							de_jure_liege_or_above = PREVPREV

									province = PREVPREV 
								}
							}
